```python
import hashlib, json, sqlite3
from pathlib import Path
from dataclasses import dataclass

CACHE_DB = Path(".cherenkov/cache/cache.db")
CACHE_TEST_DIR = Path(".cherenkov/cache/tests")

@dataclass
class CacheEntry:
    endpoint_hash: str
    spec_content_hash: str
    model_name: str
    model_version: str
    generated_test_path: str
    created_at: str

class EndpointCache:
    """SQLite-backed cache mapping endpoint spec hashes → generated test paths."""

    def __init__(self, db_path: Path = CACHE_DB):
        self.db_path = db_path
        self.db_path.parent.mkdir(parents=True, exist_ok=True)
        self._init_db()

    def _init_db(self):
        with sqlite3.connect(self.db_path) as conn:
            conn.execute("""
                CREATE TABLE IF NOT EXISTS endpoint_cache (
                    endpoint_hash TEXT PRIMARY KEY,
                    spec_content_hash TEXT NOT NULL,
                    model_name TEXT NOT NULL,
                    model_version TEXT NOT NULL,
                    generated_test_path TEXT NOT NULL,
                    created_at TEXT NOT NULL
                )
            """)

    def compute_hash(self, path: str, method: str, schema_block: dict,
                     model_name: str, model_version: str) -> str:
        """Hash = SHA-256 of (path + method + schema_json + model_name + model_version)."""
        payload = json.dumps({
            "path": path,
            "method": method,
            "schema": schema_block,
            "model": model_name,
            "version": model_version,
        }, sort_keys=True)
        return hashlib.sha256(payload.encode()).hexdigest()

    def get(self, endpoint_hash: str) -> CacheEntry | None:
        with sqlite3.connect(self.db_path) as conn:
            row = conn.execute(
                "SELECT * FROM endpoint_cache WHERE endpoint_hash = ?",
                (endpoint_hash,)
            ).fetchone()
        if row:
            return CacheEntry(*row)
        return None

    def put(self, entry: CacheEntry) -> None:
        with sqlite3.connect(self.db_path) as conn:
            conn.execute("""
                INSERT OR REPLACE INTO endpoint_cache VALUES (?, ?, ?, ?, ?, ?)
            """, (
                entry.endpoint_hash, entry.spec_content_hash,
                entry.model_name, entry.model_version,
                entry.generated_test_path, entry.created_at,
            ))

    def stats(self) -> dict:
        with sqlite3.connect(self.db_path) as conn:
            total = conn.execute("SELECT COUNT(*) FROM endpoint_cache").fetchone()[0]
        return {"cached_endpoints": total, "cache_db": str(self.db_path)}
```

### 2. Modify `cherenkov/stages/generate.py`
In `GenerateStage.run()`, wrap the LLM call with a cache check:

```python
from cherenkov.cache.endpoint_cache import EndpointCache, CacheEntry
from datetime import datetime, timezone

class GenerateStage:
    def run(self, plan_output: PlanOutput, config: Config,
            no_cache: bool = False) -> GenerateOutput:
        cache = EndpointCache() if not no_cache else None
        generated = []
        cache_hits = 0
        cache_misses = 0

        for scenario in plan_output.scenarios:
            endpoint_hash = self._compute_endpoint_hash(scenario, config)

            if cache and (entry := cache.get(endpoint_hash)):
                # Cache hit — reuse existing test file
                test_code = Path(entry.generated_test_path).read_text()
                generated.append(GeneratedTest(
                    scenario=scenario,
                    test_code=test_code,
                    from_cache=True,
                ))
                cache_hits += 1
                continue

            # Cache miss — call LLM
            test_code = self._generate_via_llm(scenario, config)
            cache_misses += 1

            if cache:
                test_path = CACHE_TEST_DIR / f"{endpoint_hash}.spec.ts"
                test_path.parent.mkdir(parents=True, exist_ok=True)
                test_path.write_text(test_code)
                cache.put(CacheEntry(
                    endpoint_hash=endpoint_hash,
                    spec_content_hash=endpoint_hash,
                    model_name=config.model_name,
                    model_version=config.model_version or "unknown",
                    generated_test_path=str(test_path),
                    created_at=datetime.now(timezone.utc).isoformat(),
                ))
            generated.append(GeneratedTest(scenario=scenario, test_code=test_code))

        return GenerateOutput(
            tests=generated,
            cache_stats={"hits": cache_hits, "misses": cache_misses},
        )

    def _compute_endpoint_hash(self, scenario: Scenario, config: Config) -> str:
        cache = EndpointCache()
        return cache.compute_hash(
            path=scenario.path,
            method=scenario.method,
            schema_block=scenario.raw_schema,
            model_name=config.model_name,
            model_version=config.model_version or "unknown",
        )
```

```python
from pathlib import Path

class PlaywrightRunner:
    def run(

        timeout_ms: int = 30_000,
```

### 2. Create `cherenkov/execution/emitters/html_report.py`

```python
from pathlib import Path
from jinja2 import Environment, FileSystemLoader
from cherenkov.core.contracts import DivergenceReport
from datetime import datetime

TEMPLATE_DIR = Path(__file__).parent
TEMPLATE_NAME = "report_template.html"

class HTMLReportEmitter:
    def emit(self, report: DivergenceReport, output_path: Path) -> Path:
        env = Environment(loader=FileSystemLoader(str(TEMPLATE_DIR)), autoescape=False)
        template = env.get_template(TEMPLATE_NAME)

        html = template.render(
            spec_title=report.spec_title or "API Conformance",
            run_time=datetime.now().strftime("%Y-%m-%d %H:%M"),
            target_url=report.target_url or "—",
            passed=report.passed,
            failed=report.failed,
            drift_count=report.drift_count,
            skipped=report.skipped,
            drift_findings=self._format_findings(report),
            endpoints=self._format_endpoints(report),
        )
        output_path.write_text(html, encoding="utf-8")
        return output_path

    def _format_findings(self, report: DivergenceReport) -> list[dict]:
        findings = []
        for f in getattr(report, "findings", []):
            findings.append({
                "endpoint": f.endpoint,
                "method": f.http_method,
                "severity": getattr(f, "severity", "medium"),
                "expected": str(f.expected),
                "actual": str(f.actual),
                "full_detail": f.summary,
            })
        return findings

    def _format_endpoints(self, report: DivergenceReport) -> list[dict]:
        endpoints = []
        for e in getattr(report, "endpoint_results", []):
            status_class = "pass" if e.passed else ("drift" if e.has_drift else "fail")
            endpoints.append({
                "path": e.path,
                "method": e.method,
                "status": "PASS" if e.passed else ("DRIFT" if e.has_drift else "FAIL"),
                "status_class": status_class,
                "suggestions": getattr(e, "tightening_suggestions", []),
            })
        return endpoints
```

### 1. Create `cherenkov/diff/` package

**`cherenkov/diff/spec_differ.py`**:
```python
from dataclasses import dataclass, field
from enum import Enum
from typing import Any
import yaml, json

class ChangeType(str, Enum):
    REMOVED_ENDPOINT = "removed_endpoint"
    REMOVED_REQUIRED_FIELD = "removed_required_field"
    NARROWED_RESPONSE_SCHEMA = "narrowed_response_schema"
    CHANGED_PARAMETER_TYPE = "changed_parameter_type"
    ADDED_ENDPOINT = "added_endpoint"
    ADDED_OPTIONAL_FIELD = "added_optional_field"
    CHANGED_DESCRIPTION = "changed_description"

@dataclass
class SpecChange:
    change_type: ChangeType
    breaking: bool
    endpoint: str
    method: str | None
    detail: str
    before: Any = None
    after: Any = None

@dataclass
class SpecDiffReport:
    breaking: list[SpecChange] = field(default_factory=list)
    additive: list[SpecChange] = field(default_factory=list)
    informational: list[SpecChange] = field(default_factory=list)

    @property
    def has_breaking_changes(self) -> bool:
        return len(self.breaking) > 0

class SpecDiffer:
    def diff(self, before_path: str, after_path: str) -> SpecDiffReport:
        before = self._load(before_path)
        after = self._load(after_path)
        report = SpecDiffReport()

        before_paths = set(before.get("paths", {}).keys())
        after_paths = set(after.get("paths", {}).keys())

        # Removed endpoints (BREAKING)
        for p in before_paths - after_paths:
            for method in before["paths"][p]:
                if method.upper() in {"GET","POST","PUT","PATCH","DELETE","OPTIONS","HEAD"}:
                    report.breaking.append(SpecChange(
                        change_type=ChangeType.REMOVED_ENDPOINT,
                        breaking=True,
                        endpoint=p, method=method.upper(),
                        detail=f"Endpoint {method.upper()} {p} was removed",
                    ))

        # Added endpoints (ADDITIVE)
        for p in after_paths - before_paths:
            for method in after["paths"][p]:
                if method.upper() in {"GET","POST","PUT","PATCH","DELETE","OPTIONS","HEAD"}:
                    report.additive.append(SpecChange(
                        change_type=ChangeType.ADDED_ENDPOINT,
                        breaking=False,
                        endpoint=p, method=method.upper(),
                        detail=f"New endpoint {method.upper()} {p}",
                    ))

        # Changed endpoints — check schemas
        for p in before_paths & after_paths:
            self._diff_endpoint(
                before["paths"][p], after["paths"][p], p, report
            )

        return report

    def _diff_endpoint(self, before_item: dict, after_item: dict,
                       path: str, report: SpecDiffReport) -> None:
        for method in before_item:
            if method not in after_item:
                continue
            before_op = before_item[method]
            after_op = after_item[method]
            self._diff_response_schemas(before_op, after_op, path, method, report)
            self._diff_parameters(before_op, after_op, path, method, report)

    def _diff_response_schemas(self, before_op, after_op, path, method, report):
        # Compare response schemas for removals / narrowing
        before_resp = before_op.get("responses", {})
        after_resp = after_op.get("responses", {})
        for status_code in before_resp:
            if status_code not in after_resp:
                report.breaking.append(SpecChange(
                    change_type=ChangeType.NARROWED_RESPONSE_SCHEMA,
                    breaking=True,
                    endpoint=path, method=method.upper(),
                    detail=f"Response {status_code} removed from {method.upper()} {path}",
                ))

    def _diff_parameters(self, before_op, after_op, path, method, report):
        before_params = {p["name"]: p for p in before_op.get("parameters", [])}
        after_params = {p["name"]: p for p in after_op.get("parameters", [])}
        for name, param in before_params.items():
            if param.get("required") and name not in after_params:
                report.breaking.append(SpecChange(
                    change_type=ChangeType.REMOVED_REQUIRED_FIELD,
                    breaking=True,
                    endpoint=path, method=method.upper(),
                    detail=f"Required parameter '{name}' removed from {method.upper()} {path}",
                ))

    def _load(self, path: str) -> dict:
        with open(path) as f:
            return yaml.safe_load(f) if path.endswith((".yaml", ".yml")) else json.load(f)
```

```python
import hashlib, json
from pathlib import Path

def _check_spec_changed(spec_path: str) -> bool:
    cache_file = Path(".cherenkov/spec_hash.json")
    current_hash = hashlib.sha256(open(spec_path, "rb").read()).hexdigest()
    if cache_file.exists():
        stored = json.loads(cache_file.read_text())
        last_hash = stored.get("hash")
        if last_hash != current_hash:
            print(f"⚠️  Spec has changed since last run. Run `cherenkov diff` for details.")
            return True
    cache_file.parent.mkdir(exist_ok=True)
    cache_file.write_text(json.dumps({"hash": current_hash, "spec": spec_path}))
    return False
```

```python
import xml.etree.ElementTree as ET
from pathlib import Path
from cherenkov.sources.accessibility.contracts import PageTarget, AccessibilityScenario

class AccessibilitySourceAdapter:
    """Parses a URL list or sitemap.xml → list of PageTarget objects."""

    def __init__(self, spec_path: str):
        self.spec_path = spec_path
        self.pages = self._load_pages()

    def _load_pages(self) -> list[PageTarget]:
        path = Path(self.spec_path)
        if path.suffix == ".xml":
            return self._parse_sitemap(path)
        elif path.suffix in (".txt", ".list"):
            return self._parse_url_list(path)
        else:
            raise ValueError(f"Unsupported spec format for accessibility: {path.suffix}. Use sitemap.xml or urls.txt")

    def _parse_sitemap(self, path: Path) -> list[PageTarget]:
        tree = ET.parse(path)
        root = tree.getroot()
        ns = {"sm": "http://www.sitemaps.org/schemas/sitemap/0.9"}
        pages = []
        for url_elem in root.findall("sm:url", ns):
            loc = url_elem.find("sm:loc", ns)
            if loc is not None:
                pages.append(PageTarget(url=loc.text.strip()))
        return pages

    def _parse_url_list(self, path: Path) -> list[PageTarget]:
        return [
            PageTarget(url=line.strip())
            for line in path.read_text().splitlines()
            if line.strip() and not line.startswith("#")
        ]

    def iter_scenarios(self) -> list[AccessibilityScenario]:
        scenarios = []
        for page in self.pages:
            scenarios.append(AccessibilityScenario(
                page=page,
                scenario_type="wcag_aa_full",
                description=f"WCAG 2.1 AA full scan of {page.url}",
            ))
        return scenarios
```
